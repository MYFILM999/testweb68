Notes: SASS / Utility API (ไม่ใช่โค้ด HTML)

// Override Bootstrap SASS Variables ก่อน @import
$primary:    #16a34a;
$secondary:  #0ea5e9;
$success:    #22c55e;
$danger:     #ef4444;

$font-family-base: 'Sarabun', sans-serif;
$border-radius:    0.75rem;
$border-radius-lg: 1rem;

$enable-shadows:   true;
$enable-negative-margins: true;

// Import Bootstrap หลังจาก Override Variables
@import "bootstrap/scss/bootstrap";

// ติดตั้ง
npm install bootstrap@5.3 sass

// Compile
npx sass custom.scss output.css --style compressed --watch

// Utility API example
$utilities: map-merge(
  $utilities,
  (
    "cursor": (
      property: cursor,
      class: cursor,
      values: auto default pointer grab
    )
  )
);

@import "bootstrap/scss/utilities/api";
